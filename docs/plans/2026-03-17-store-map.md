### Task 2: Create StoreMap component

**Files:**
- Create: `web/components/StoreMap.tsx`

**Step 1: Create the component**

```tsx
"use client";
import { GoogleMap, LoadScript, Marker, InfoWindow } from "@react-google-maps/api";
import { useState } from "react";

interface Store {
  id: string;
  name: string;
  address: string;
  lat: number;
  lng: number;
}

const stores: Store[] = [
  {
    id: "shoprite-queens",
    name: "ShopRite Queens",
    address: "ShopRite Queens, 11356",
    lat: 40.7844,
    lng: -73.8441,
  },
  {
    id: "shoprite-brooklyn",
    name: "ShopRite Brooklyn",
    address: "ShopRite Brooklyn, 11230",
    lat: 40.6195,
    lng: -73.9584,
  },
];

const containerStyle = { width: "100%", height: "100%" };

const center = {
  lat: (stores[0].lat + stores[1].lat) / 2,
  lng: (stores[0].lng + stores[1].lng) / 2,
};

export default function StoreMap() {
  const [selected, setSelected] = useState<Store | null>(null);
  const apiKey = process.env.NEXT_PUBLIC_GOOGLE_MAPS_API_KEY;

  if (!apiKey) {
    return (
      <div className="w-full h-full rounded-2xl flex flex-col items-center justify-center"
        style={{ background: "linear-gradient(135deg, #E8F5E9 0%, #C8E6C9 40%, #A5D6A7 100%)" }}>
        <svg width="22" height="22" viewBox="0 0 24 24" fill="none" stroke="rgba(45,106,79,0.5)" strokeWidth="1.5">
          <path d="M21 10c0 7-9 13-9 13s-9-6-9-13a9 9 0 0 1 18 0z"/>
          <circle cx="12" cy="10" r="3"/>
        </svg>
        <span className="text-xs mt-1 font-medium" style={{ color: "rgba(45,106,79,0.7)" }}>
          Set NEXT_PUBLIC_GOOGLE_MAPS_API_KEY to enable map
        </span>
      </div>
    );
  }

  return (
    <LoadScript googleMapsApiKey={apiKey}>
      <GoogleMap mapContainerStyle={containerStyle} center={center} zoom={11}>
        {stores.map((store) => (
          <Marker
            key={store.id}
            position={{ lat: store.lat, lng: store.lng }}
            onClick={() => setSelected(store)}
          />
        ))}
        {selected && (
          <InfoWindow
            position={{ lat: selected.lat, lng: selected.lng }}
            onCloseClick={() => setSelected(null)}
          >
            <div>
              <strong>{selected.name}</strong>
              <br />
              <span style={{ fontSize: 12 }}>{selected.address}</span>
            </div>
          </InfoWindow>
        )}
      </GoogleMap>
    </LoadScript>
  );
}
```
